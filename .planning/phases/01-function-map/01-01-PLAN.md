1. Create `get-shit-done/bin/lib/fmap.cjs`:

```javascript
const fs = require('fs');
const path = require('path');
const { output, error, planningRoot, toPosixPath } = require('./core.cjs');

const FMAP_FILENAME = 'function-map.json';

function fmapPath(cwd) {
  return path.join(planningRoot(cwd), FMAP_FILENAME);
}

function normalizeKey(key) {
  // Strip leading ./ and normalize to POSIX slashes
  let k = key.replace(/\\/g, '/');
  if (k.startsWith('./')) k = k.slice(2);
  return k;
}

function readMap(cwd) {
  const p = fmapPath(cwd);
  if (!fs.existsSync(p)) return {};
  return JSON.parse(fs.readFileSync(p, 'utf-8'));
}

function writeMap(cwd, map) {
  const p = fmapPath(cwd);
  fs.writeFileSync(p, JSON.stringify(map, null, 2), 'utf-8');
}
```

Implement these exported functions:
- `cmdFmapGet(cwd, key, raw)` -- If no key: `output(readMap(cwd), raw)`. If key: normalize key, lookup in map, output entry or `error('Function Map: key not found: <key>')`.
- `cmdFmapUpdate(cwd, args, raw)` -- Parse `--data <json>` from args. Parse `--replace-file <filepath>` flag. If replace-file: delete all keys starting with `normalizeKey(filepath) + '::'` before merging. Merge patch via `Object.assign(map, patch)`. Write map. Output `{ updated: N, total: M }`.
- `cmdFmapStats(cwd, raw)` -- Read map, count entries, group by `kind` field. Output `{ total, by_kind, path: fmapPath(cwd) }`.
- `cmdFmapFullScan(cwd, raw)` -- Output `{ action: 'full-scan', message: 'Trigger gsd-cataloger agent for full rescan' }`.

Module exports: `module.exports = { cmdFmapGet, cmdFmapUpdate, cmdFmapStats, cmdFmapFullScan };`
